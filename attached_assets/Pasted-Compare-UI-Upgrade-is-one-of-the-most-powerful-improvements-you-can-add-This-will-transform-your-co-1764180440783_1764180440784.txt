import CompareTechTag from "./CompareTechTag";
import { getCategoryIcon } from "./icons/CategoryIcons";

export default function CompareCategoryCard({ title, shared = [], aOnly = [], bOnly = [] }) {
  const Icon = getCategoryIcon(title);

  if (shared.length === 0 && aOnly.length === 0 && bOnly.length === 0) return null;

  return (
    <div className="bg-white border rounded-xl p-6 shadow-sm">
      
      <div className="flex items-center gap-2 mb-4">
        <Icon className="w-6 h-6 text-blue-600" />
        <h3 className="text-xl font-semibold">{title}</h3>
      </div>

      <div className="grid grid-cols-1 md:grid-cols-3 gap-4">

        {/* Unique to A */}
        <div>
          <h4 className="font-semibold mb-2 text-blue-600">Unique to A</h4>
          <div className="flex flex-wrap gap-2">
            {aOnly.map((tech, i) => (
              <CompareTechTag key={i} name={tech} type="uniqueA" />
            ))}
          </div>
        </div>

        {/* Shared */}
        <div>
          <h4 className="font-semibold mb-2 text-green-600">Shared</h4>
          <div className="flex flex-wrap gap-2">
            {shared.map((tech, i) => (
              <CompareTechTag key={i} name={tech} type="shared" />
            ))}
          </div>
        </div>

        {/* Unique to B */}
        <div>
          <h4 className="font-semibold mb-2 text-purple-600">Unique to B</h4>
          <div className="flex flex-wrap gap-2">
            {bOnly.map((tech, i) => (
              <CompareTechTag key={i} name={tech} type="uniqueB" />
            ))}
          </div>
        </div>

      </div>
    </div>
  );
}


This gives us a beautiful 3-column comparison for each category.

"use client";

import { useSearchParams } from "next/navigation";
import { useEffect, useState } from "react";
import CompareCategoryCard from "@/components/CompareCategoryCard";

export default function ComparePage() {
  const params = useSearchParams();
  const urlA = params.get("a");
  const urlB = params.get("b");

  const [dataA, setDataA] = useState(null);
  const [dataB, setDataB] = useState(null);
  const [loading, setLoading] = useState(true);

  const apiBase = process.env.NEXT_PUBLIC_API_BASE;

  useEffect(() => {
    if (!urlA || !urlB) return;

    Promise.all([
      fetch(`${apiBase}/scan?url=${encodeURIComponent(urlA)}`).then(res => res.json()),
      fetch(`${apiBase}/scan?url=${encodeURIComponent(urlB)}`).then(res => res.json())
    ]).then(([a, b]) => {
      setDataA(a);
      setDataB(b);
      setLoading(false);
    });
  }, [urlA, urlB]);

  if (!urlA || !urlB) {
    return <div className="p-10 text-center text-red-500">Missing URLs.</div>;
  }

  if (loading) {
    return <div className="p-10 text-center">Comparing sites‚Ä¶</div>;
  }

  if (!dataA || !dataB) {
    return <div className="p-10 text-center text-red-600">Comparison data missing.</div>;
  }

  // Extract raw technologies
  const techA = dataA.technologies || [];
  const techB = dataB.technologies || [];

  // Shared vs unique
  const shared = techA.filter(t => techB.includes(t));
  const aOnly = techA.filter(t => !techB.includes(t));
  const bOnly = techB.filter(t => !techA.includes(t));

  // Convert to categories (you already have categorizeTechnologies function)
  const categorize = (list) => {
    // reuse your existing categorizeTechnologies() here
    // for now assume list is simple
    return {}; 
  };

  const categories = ["Frontend", "Backend", "CMS", "Hosting", "CDN", "Analytics", "Marketing", "Security", "Payments", "Other"];

  return (
    <div className="max-w-6xl mx-auto px-6 py-10">
      <h1 className="text-3xl font-bold mb-8 text-center">
        Compare Technologies:  
        <span className="text-blue-600"> {urlA} </span>  
        vs  
        <span className="text-purple-600"> {urlB} </span>
      </h1>

      {/* Summary stats */}
      <div className="bg-white border rounded-xl p-6 shadow-sm mb-10 flex justify-center gap-10 text-center">
        <div>
          <div className="text-3xl font-bold text-green-600">{shared.length}</div>
          <div className="text-slate-600">Shared Tech</div>
        </div>
        <div>
          <div className="text-3xl font-bold text-blue-600">{aOnly.length}</div>
          <div className="text-slate-600">A Only</div>
        </div>
        <div>
          <div className="text-3xl font-bold text-purple-600">{bOnly.length}</div>
          <div className="text-slate-600">B Only</div>
        </div>
      </div>

      {/* Per-category comparison */}
      <div className="flex flex-col gap-8">
        {categories.map(cat => (
          <CompareCategoryCard
            key={cat}
            title={cat}
            shared={shared.filter(t => t.toLowerCase().includes(cat.toLowerCase()))}
            aOnly={aOnly.filter(t => t.toLowerCase().includes(cat.toLowerCase()))}
            bOnly={bOnly.filter(t => t.toLowerCase().includes(cat.toLowerCase()))}
          />
        ))}
      </div>
    </div>
  );
}
