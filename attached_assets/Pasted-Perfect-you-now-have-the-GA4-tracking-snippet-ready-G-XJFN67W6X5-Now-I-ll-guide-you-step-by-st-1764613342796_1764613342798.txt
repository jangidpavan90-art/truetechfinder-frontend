⭐ STEP 1 — Add GA4 Using next/script in app/layout.js

Open app/layout.js

Inside <head>, add the async script

Add a second Script tag for the config

Here is the exact code you must insert (replace your empty placeholder):

✅ FINAL, CORRECT GA4 IMPLEMENTATION (copy/paste)
// app/layout.js
import "./globals.css";
import Footer from "@/components/Footer";
import Script from "next/script";

export const metadata = {
  title: "TrueTechFinder",
  description: "Find what powers any website — instantly."
};

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <head>
        {/* Favicon */}
        <link rel="icon" href="/favicon.svg" />
        <link rel="apple-touch-icon" sizes="192x192" href="/logo-192.png" />
        <meta name="theme-color" content="#2563EB" />

        {/* Google Analytics */}
        <Script
          src="https://www.googletagmanager.com/gtag/js?id=G-XJFN67W6X5"
          strategy="afterInteractive"
        />

        <Script id="ga-init" strategy="afterInteractive">
          {`
            window.dataLayer = window.dataLayer || [];
            function gtag(){dataLayer.push(arguments);}
            gtag('js', new Date());
            gtag('config', 'G-XJFN67W6X5');
          `}
        </Script>
      </head>

      <body className="bg-white text-slate-900">
        {children}
        <Footer />
      </body>
    </html>
  );
}
