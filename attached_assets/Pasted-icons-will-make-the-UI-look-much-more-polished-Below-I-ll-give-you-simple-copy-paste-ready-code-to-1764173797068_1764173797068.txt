/* --- mapping function --- */
const mapping = [
  { test: /react/i, icon: IconReact },
  { test: /next/i, icon: IconNext },
  { test: /wordpress|wp-engine|wpcom|wp-admin/i, icon: IconWordPress },
  { test: /cloudflare/i, icon: IconCloudflare },
  { test: /vercel|now|vercel\.app/i, icon: IconAws },
  { test: /aws|amazon/i, icon: IconAws },
  { test: /nginx/i, icon: IconNginx },
  // add more rules as you need
];

export function getTechIcon(techName) {
  if (!techName) return IconFallback;
  for (const m of mapping) {
    if (m.test.test(String(techName))) return m.icon;
  }
  return IconFallback;
}

2) Create a TechTag component that uses icons

File: components/TechTag.js

// components/TechTag.js
import React from "react";
import { getTechIcon } from "./icons/TechIcons";

export default function TechTag({ name }) {
  const Icon = getTechIcon(name);

  return (
    <div className="inline-flex items-center gap-2 px-3 py-1 bg-white border rounded-full text-sm shadow-sm">
      <span className="text-slate-600">
        <Icon className="w-4 h-4" />
      </span>
      <span className="text-sm text-slate-700">{name}</span>
    </div>
  );
}

3) Update TechCategoryCard to use TechTag

Replace your existing components/TechCategoryCard.js with this:

// components/TechCategoryCard.js
import TechTag from "./TechTag";

export default function TechCategoryCard({ title, items = [] }) {
  if (!items || items.length === 0) return null;

  return (
    <div className="bg-white border rounded-xl p-6 shadow-sm hover:shadow-md transition">
      <h3 className="text-xl font-semibold mb-4">{title}</h3>

      <div className="flex flex-wrap gap-3">
        {items.map((item, i) => (
          <TechTag key={i} name={item} />
        ))}
      </div>
    </div>
  );
}
