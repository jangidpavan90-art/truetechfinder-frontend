‚≠ê STEP 1 ‚Äî Create Screenshot & Favicon Component

üëâ Create file:

components/SitePreview.js

Paste this:

"use client";

import { useEffect, useState } from "react";

export default function SitePreview({ url }) {
  const [favicon, setFavicon] = useState("");
  const [screenshot, setScreenshot] = useState("");

  const domain = url.replace("https://", "").replace("http://", "").split("/")[0];

  // Favicon
  useEffect(() => {
    const iconUrl = `https://www.google.com/s2/favicons?domain=${domain}&sz=128`;
    setFavicon(iconUrl);
  }, [domain]);

  // Screenshot (lazy load)
  useEffect(() => {
    const imgUrl = `https://image.thum.io/get/width/1200/https://${domain}`;
    setScreenshot(imgUrl);
  }, [domain]);

  return (
    <div className="bg-white border rounded-xl p-6 shadow-sm mb-10">
      <div className="flex items-center gap-4 mb-6">
        <img
          src={favicon}
          alt="favicon"
          className="w-10 h-10 rounded-md border"
        />
        <div>
          <h2 className="text-2xl font-semibold">{domain}</h2>
          <p className="text-slate-500">{url}</p>
        </div>
      </div>

      {/* Screenshot */}
      <div className="rounded-xl overflow-hidden border shadow-sm bg-gray-50 w-full h-[300px] flex items-center justify-center">
        <img
          src={screenshot}
          alt="screenshot"
          className="w-full h-full object-cover"
        />
      </div>
    </div>
  );
}
